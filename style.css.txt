const questions = [
  {
    text: "You receive an email saying you won a gift card and must enter your password to claim it. What should you do?",
    options: [
      "Click the link and enter your password quickly.",
      "Ignore or delete the email and do not click the link.",
      "Forward it to everyone so they can win too."
    ],
    correctIndex: 1
  },
  {
    text: "A message from your 'IT department' asks you to install software from an unknown website. What is the safest response?",
    options: [
      "Install it immediately so your device stays secure.",
      "Reply with your username and password instead.",
      "Verify with your real IT support using a known contact method."
    ],
    correctIndex: 2
  },
  {
    text: "Which of these is a common sign of a phishing email?",
    options: [
      "Professional language and correct spelling.",
      "Unexpected urgency and threats if you don’t act.",
      "Sent from your own address book only."
    ],
    correctIndex: 1
  }
];

function renderQuiz() {
  const container = document.getElementById("quiz-container");
  questions.forEach((q, index) => {
    const div = document.createElement("div");
    div.className = "question";

    const title = document.createElement("h3");
    title.textContent = `Question ${index + 1}: ${q.text}`;
    div.appendChild(title);

    q.options.forEach((opt, i) => {
      const label = document.createElement("label");
      label.style.display = "block";
      label.style.marginBottom = "0.25rem";

      const input = document.createElement("input");
      input.type = "radio";
      input.name = `question-${index}`;
      input.value = i;

      label.appendChild(input);
      label.appendChild(document.createTextNode(" " + opt));
      div.appendChild(label);
    });

    container.appendChild(div);
  });
}

function gradeQuiz() {
  let score = 0;

  questions.forEach((q, index) => {
    const selected = document.querySelector(
      `input[name="question-${index}"]:checked`
    );
    if (selected && Number(selected.value) === q.correctIndex) {
      score++;
    }
  });

  const result = document.getElementById("result");
  const total = questions.length;
  const percent = Math.round((score / total) * 100);

  let message = "";
  if (percent === 100) {
    message = "Perfect! You’re a phishing-spotting pro.";
  } else if (percent >= 67) {
    message = "Nice job! You’re mostly safe, but review the tips above.";
  } else {
    message = "Be careful. Review the best practices and try again.";
  }

  result.textContent = `You scored ${score}/${total} (${percent}%). ${message}`;
}

document.addEventListener("DOMContentLoaded", () => {
  renderQuiz();
  document
    .getElementById("submit-quiz")
    .addEventListener("click", gradeQuiz);
});
